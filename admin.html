function downloadLocationMonthlySchedule() { const month = document.getElementById("exportMonthSelect").value; const locationName = document.getElementById("exportLocationSelect").value; if (!month || !locationName) { alert("❌ 請先選擇下載月份與要統整的案場名稱！"); return; } showMask(`正在動態調閱並交叉統整 ${locationName} 總班表...`); const matches = month.match(/(\d+)年(\d+)月/); const year = parseInt(matches[1]); const monthNum = parseInt(matches[2]); const totalDays = new Date(year, monthNum, 0).getDate(); let excelHeaders = ["工號", "姓名"]; for (let d = 1; d <= totalDays; d++) { excelHeaders.push(`${d}日`); } excelHeaders.push("當月指派天數"); let excelRows = []; let loadedCount = 0; if (globalEmployees.length === 0) { hideMask(); alert("目前名冊庫無同仁資料"); return; } globalEmployees.forEach(emp => { const url = `${BASE_GAS_URL}?action=fetchSchedule&employeeId=${emp.id}&month=${month}`; sendJsonpRequest(url, "cb_exp_pack_" + emp.id + "_" + Math.floor(Math.random()*1000), function(res) { loadedCount++; if (res.success && res.schedule && res.schedule.length > 0) { let rowObj = { "工號": emp.id, "姓名": emp.name }; let hasDutyAtLocation = false; let dutyDaysCount = 0; for (let d = 1; d <= totalDays; d++) { rowObj[`${d}日`] = ""; } res.schedule.forEach(item => { let rawContent = String(item.shift).trim(); if (rawContent && rawContent.indexOf("-") !== -1) { let parts = rawContent.split("-"); if (parts[0] === locationName) { let dayNum = parseInt(String(item.date).replace("日", "").trim(), 10); rowObj[`${dayNum}日`] = parts[1] || ""; hasDutyAtLocation = true; dutyDaysCount++; } } }); if (hasDutyAtLocation) { rowObj["當月指派天數"] = dutyDaysCount + " 天"; excelRows.push(rowObj); } } if (loadedCount === globalEmployees.length) { hideMask(); // 確保不論有沒有資料，都能解鎖畫面！ if (excelRows.length === 0) { alert(`💡 查無同仁被指派在【${locationName}】案場值勤。`); return; } const worksheet = XLSX.utils.json_to_sheet(excelRows, { header: excelHeaders }); const workbook = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(workbook, worksheet, "案場班表統整"); XLSX.writeFile(workbook, `${month}_${locationName}_全人員班表統整.xlsx`); } }).catch(() => { // 異常防護 if (loadedCount === globalEmployees.length) { hideMask(); } }); }); }
